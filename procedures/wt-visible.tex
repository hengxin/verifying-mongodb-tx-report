\begin{algorithm}[H]
  \begin{algorithmic}[1]
    \Procedure{\txnvis}{$\txnvar, \tidvar$}
      \State \Return
        $\lnot \big(\tidvar = -1
        \lor \tidvar \in \txnvar.\snapshot
        \lor (\tidvar \ge \txnvar.\snapmax \land \tidvar \neq \txnvar.\tid)\big)$
    \EndProcedure
  \end{algorithmic}
\end{algorithm}
