\begin{algorithm}[H]
  \begin{algorithmic}[1]
    \Procedure{\txnvis}{$\txnvar, \tidvar, \rsstyle{\tsvar}$}
      \State \Return
          $\lnot \big(\tidvar = -1
          \lor \tidvar \in \txnvar.\snapshot
          \lor (\tidvar \ge \txnvar.\snapmax \land \tidvar \neq \txnvar.\tid)\big)$
          $\rsstyle{\land\; \big(\tsvar \neq \wttsnone \land \tsvar \le \txnvar.\readts\big)}$
    \EndProcedure
  \end{algorithmic}
\end{algorithm}
